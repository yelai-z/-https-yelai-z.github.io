# vue自定义指令用canvas渲染图片和文字

  1. 准备数据
data () {
    return {
      params: {
        url: '../static/logo.png',
        text: '文案'
      }
    }
  }
  1. 准备directives.js文件
const canvasText = {
  inserted: (el, binding) => {
    console.log(el, binding)
    let cvs = el.getContext('2d')
    let img = new Image()
    img.src = binding.value.url
    console.log(cvs, img)
    img.onload = function () {
      console.log(1)
      cvs.drawImage(img, 0, 0)
      cvs.fillStyle = 'blue'
      cvs.font = '33px Adobe Ming Std'
      cvs.fillText(binding.value.text, 80, 80)
    }
  }
}

export default {
  'canvas-text': canvasText
}
  1. 在main.js中引入自定义指令文件
import Vue from 'vue'
import App from './App'
import router from './router'
import directives from './assets/directives'

Vue.config.productionTip = false

Object.keys(directives).map(ele => {
  Vue.directive(ele, directives[ele])
})

/* eslint-disable no-new */
new Vue({
  el: '#app',
  router,
  components: { App },
  template: '<App/>'
})
  1. 在页面中准备canvas元素，添加指令
<canvas" width="200" height="200" v-canvas-text="params"></canvas>
